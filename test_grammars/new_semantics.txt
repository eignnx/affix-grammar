data Gender = masculine | feminine | nonbinary
data Person = 1st | 2nd | 3rd
data Number = singular | plural


rule want.Number.Person =
    .singular {
        .1st -> "veux"
        .2nd -> "veux"
        .3rd -> "veut"
    }
    .plural {
        .1st -> "voulons"
        .2nd -> "voulez"
        .3rd -> "voulent"
    }
