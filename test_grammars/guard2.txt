data Gender = masculine | feminine | nonbinary

rule start = noun_phrase

rule noun_phrase = name.Gender1 "est" adj.Gender1 + "."

rule name.Gender =
    .masculine -> "Bob"
    .feminine -> "Sally"
    .nonbinary -> "Jesse"

rule adj.Gender =
    .masculine -> "gentil"
    .feminine -> "gentille"
    .nonbinary -> "kind"
